FILE(GLOB SRC *.cpp *.h)

ADD_EXECUTABLE(monitor_service WIN32 ${SRC})

TARGET_LINK_LIBRARIES(monitor_service
						ryzom_gameshare
						${LIBXML2_LIBRARIES}
						${ZLIB_LIBRARY}
						${MYSQL_LIBRARIES}
						nelmisc 
						nelnet
						nelgeorges)

  ADD_NATIVE_PRECOMPILED_HEADER(monitor_service ${CMAKE_CURRENT_SOURCE_DIR}/stdpch.h ${CMAKE_CURRENT_SOURCE_DIR}/stdpch.cpp)
ENDIF(WITH_PCH)
